¿Cómo funciona?
Flujo completo en 4 pasos, desde el QR en la estación hasta la política pública
📱
1. Ingesta QR → WhatsApp
El joven escanea un QR en la estación. Se activa un Deep Link que abre WhatsApp con el chatbot IncidePe, sin descargar ninguna app.
WhatsApp APIQR CodeDeep Link
🔍
2. Pipeline semántico IA
El testimonio pasa por anonimización (Ley 29733), clasificación en ejes CEPLAN y vinculación normativa (Ley 26300) de forma automática.
Python NERRegex PIINLP
📍
3. Tracker de incidencia
El usuario recibe un link único con un tracker estilo "delivery" que muestra en qué estado está su aporte en tiempo real.
PWAWebSocketREST API
🏆
4. Ranking por Octógonos
El IAJ de cada institución se calcula con 4 indicadores. El octógono se activa en GOB.PE como sello verde (logro) o rojo (sanción).
DashboardIAJ FormulaGOB.PE API

# incidepe/anonymizer.py # MIT License — Hackathon TransformaGob 2026 # Anonimización bajo Ley N.º 29733 Art. 13 import re from typing import TypedDict, List class PIIRemoval(TypedDict): type: str count: int class AnonResult(TypedDict): anon_text: str pii_removed: List[PIIRemoval] law_compliance: str PATTERNS: dict[str, re.Pattern] = { "phone": re.compile(r'\b(\+51|0051)?[\s\-]?9\d{8}\b'), "dni": re.compile(r'\b\d{8}\b'), "ruc": re.compile(r'\b(10|20)\d{9}\b'), "address": re.compile(r'\b(jr\.?|av\.?|calle|mz\.?|lt\.?)\s+\w+', re.IGNORECASE), "email": re.compile(r'\b[\w._%+\-]+@[\w.\-]+\.[a-zA-Z]{2,}\b'), } def anonymize(raw_text: str) -> AnonResult: """Elimina PII del testimonio según Ley N.º 29733.""" cleaned = raw_text removals: List[PIIRemoval] = [] for label, pattern in PATTERNS.items(): matches = pattern.findall(cleaned) if matches: removals.append({"type": label, "count": len(matches)}) cleaned = pattern.sub( f"[{label.upper()}_REDACTED]", cleaned ) return { "anon_text": cleaned, "pii_removed": removals, "law_compliance": "Ley N.º 29733 Art. 13 — Datos Sensibles", }
# incidepe/classifier.py # Clasificación semántica por ejes CEPLAN # MIT License — Hackathon TransformaGob 2026 from typing import TypedDict EJES: dict[str, list[str]] = { "Transporte y Entorno Seguro": [ "metro","bus","combi","tráfico","pista","semáforo", "accidente","mototaxi","paradero","seguridad vial","ruta"], "Empleo y Oportunidades": [ "trabajo","chamba","sueldo","empleo","contrato", "prácticas","desempleo","empresa","freelance","oferta"], "Seguridad Ciudadana": [ "robo","delincuencia","asalto","policía","serenazgo", "cámara","inseguridad","pandilla","vigilancia"], "Salud Mental y Bienestar": [ "estrés","ansiedad","depresión","psicólogo","salud", "hospital","centro de salud","medicamento","terapia"], "Educación y Capacitación": [ "universidad","instituto","beca","estudiar","colegio", "capacitación","taller","aprender","curso","egresado"], } class ClassResult(TypedDict): eje: str confidence: float all_scores: dict ceplan_fase: str ley_26300_cat: str def classify_testimony(anon_text: str) -> ClassResult: """Mapea testimonio a ejes temáticos CEPLAN con score de confianza.""" scores = {} text_lower = anon_text.lower() for eje, keywords in EJES.items(): hits = sum(1 for kw in keywords if kw in text_lower) if hits > 0: scores[eje] = round(hits / len(keywords), 3) if not scores: return { "eje": "Sin clasificar", "confidence": 0.0, "all_scores": {}, "ceplan_fase": "Pendiente revisión manual", "ley_26300_cat": "Pendiente" } best = max(scores, key=scores.get) return { "eje": best, "confidence": scores[best], "all_scores": scores, "ceplan_fase": "Fase 1 — Diagnóstico del Problema Público", "ley_26300_cat": "Consulta Ciudadana Virtual Masiva", }
# incidepe/iaj_calculator.py # Índice de Apertura Juvenil + Sistema de Octógonos # MIT License — Hackathon TransformaGob 2026 from dataclasses import dataclass from enum import Enum class Octogono(Enum): VERDE = "VERDE: ALTO EN PARTICIPACIÓN EFECTIVA" ROJO_A = "ROJO: EXCESO DE ADULTOCENTRISMO" ROJO_B = "ROJO: BAJO EN TRAZABILIDAD" AMBAR = "ÁMBAR: EN PROCESO DE MEJORA" @dataclass class IAJResult: tap: float itd: float dpp: float gsc: float iaj: float = 0.0 octogono: Octogono = Octogono.AMBAR certificado_joven: bool = False razon: str = "" def calcular_iaj( tap: float, itd: float, dpp: float, gsc: float ) -> IAJResult: """ Fórmula oficial IAJ: IAJ = (TAP × 0.40) + (ITD × 0.20) + (DPP × 0.20) + (GSC × 0.20) Todos los valores en porcentaje (0 a 100). """ iaj = (tap * 0.40) + (itd * 0.20) + (dpp * 0.20) + (gsc * 0.20) result = IAJResult(tap=tap, itd=itd, dpp=dpp, gsc=gsc, iaj=round(iaj, 2)) # Reglas de activación — orden de prioridad estricto if itd < 30: result.octogono = Octogono.ROJO_B result.razon = "ITD < 30%: sanción por no devolver resultados" elif iaj < 50 and tap < 15: result.octogono = Octogono.ROJO_A result.razon = "IAJ < 50% y TAP < 15%: adultocentrismo activo" elif iaj >= 85: result.octogono = Octogono.VERDE result.certificado_joven = True result.razon = "Certificado Joven otorgado — destacado en GOB.PE" else: result.octogono = Octogono.AMBAR result.razon = "En proceso — mejorar TAP e ITD para alcanzar el verde" return result
# incidepe/pipeline.py # Orquestador principal del sistema IncidePe # MIT License — Hackathon TransformaGob 2026 import uuid, datetime from .anonymizer import anonymize from .classifier import classify_testimony MACROZONAS = { "QR_BAYOVAR": "Lima Este", "QR_VES": "Lima Sur", } def run_pipeline( raw_text: str, estacion_id: str, ) -> dict: """ Pipeline completo: 1. Anonimiza PII (Ley 29733) 2. Clasifica por eje CEPLAN 3. Registra en macrozona correspondiente 4. Devuelve payload listo para tracker """ macrozona = MACROZONAS.get(estacion_id, "Lima Metropolitana") anon = anonymize(raw_text) cls = classify_testimony(anon["anon_text"]) year = datetime.date.today().year uid = str(uuid.uuid4())[:6].upper() return { "incidencia_id": f"INC-{year}-{estacion_id.replace('QR_','')}-{uid}", "timestamp": datetime.datetime.utcnow().isoformat() + "Z", "estacion": estacion_id, "macrozona": macrozona, "anonimizacion": anon, "clasificacion": cls, "estado": 1, "ley_26300_cat": cls["ley_26300_cat"], "tracker_url": f"https://incidepe.gob.pe/tracker/INC-{year}-{uid}", } # --- Uso de ejemplo --- if __name__ == "__main__": resultado = run_pipeline( raw_text="Los buses de la ruta 48 nunca paran en el paradero", estacion_id="QR_VES", ) import json print(json.dumps(resultado, ensure_ascii=False, indent=2))
