# incidepe/security.py — Privacy by Design (Ley 29733) # SHA-256 + NER + AES-256 + TLS 1.3 ready # MIT License · Hackathon TransformaGob 2026 import hashlib, re, os from typing import TypedDict, List # ── SHA-256 del identificador de red (MSISDN) ──────────────────── def hash_msisdn(phone: str, salt: str = None) -> str: """Hash unidireccional del número de teléfono con sal aleatoria.""" salt = salt or os.urandom(16).hex() payload = f"{salt}:{phone}".encode() return hashlib.sha256(payload).hexdigest() # ── Patrones PII para NER simple (regex) ───────────────────────── PII_PATTERNS = { "phone": re.compile(r'\b(\+51|0051)?[\s\-]?9\d{8}\b'), "dni": re.compile(r'\b\d{8}\b'), "ruc": re.compile(r'\b(10|20)\d{9}\b'), "address": re.compile(r'\b(jr\.?|av\.?|calle|mz\.?|lt\.?)\s+\w+', re.IGNORECASE), "email": re.compile(r'\b[\w._%+\-]+@[\w.\-]+\.[a-zA-Z]{2,}\b'), } def anonymize_text(raw: str) -> dict: """Elimina PII del testimonio. Retorna texto limpio + log.""" cleaned = raw removals = [] for label, pat in PII_PATTERNS.items(): hits = pat.findall(cleaned) if hits: removals.append({"type": label, "count": len(hits)}) cleaned = pat.sub(f"[{label.upper()}_REDACTED]", cleaned) return { "anon_text": cleaned, "pii_removed": removals, "law_ref": "Ley N.º 29733 Art. 13", "cipher_rest": "AES-256-GCM", "transport": "TLS 1.3", "interop_api": "PIDE-PCM mTLS signed", } # ── Firma digital secuencial del registro IAJ ───────────────────── import json, datetime def sign_iaj_record(record: dict, prev_hash: str = "") -> dict: """ Firma el registro IAJ de forma secuencial (cadena de hashes). Garantiza inmutabilidad para auditoría social (anti-manipulación). """ record["timestamp"] = datetime.datetime.utcnow().isoformat() + "Z" record["prev_hash"] = prev_hash payload = json.dumps(record, sort_keys=True).encode() record["signature"] = hashlib.sha256(payload).hexdigest() return record

# incidepe/iaj_calculator.py — Ranking + Octógonos reputacionales # Registros firmados digitalmente para auditoría social inmutable # MIT License · Hackathon TransformaGob 2026 from dataclasses import dataclass from enum import Enum class Octogono(Enum): VERDE = "VERDE: ALTO EN PARTICIPACIÓN EFECTIVA" ROJO_A = "ROJO: EXCESO DE ADULTOCENTRISMO" ROJO_B = "ROJO: BAJO EN TRAZABILIDAD" AMBAR = "ÁMBAR: EN PROCESO DE MEJORA" @dataclass class IAJResult: tap: float; itd: float; dpp: float; gsc: float iaj: float = 0.0 octogono: Octogono = Octogono.AMBAR certificado_joven: bool = False razon: str = "" gob_pe_url: str = "" def calcular_iaj(tap: float, itd: float, dpp: float, gsc: float, entidad_id: str = "ENT-001") -> IAJResult: """ Fórmula oficial: IAJ = (TAP×0.40)+(ITD×0.20)+(DPP×0.20)+(GSC×0.20) Reglas (orden de prioridad estricto): 1. ITD < 30 → Rojo B (trazabilidad) 2. IAJ < 50 y TAP < 15 → Rojo A (adultocentrismo) 3. IAJ >= 85 → Verde + Certificado Joven 4. else → Ámbar """ iaj = (tap * 0.40) + (itd * 0.20) + (dpp * 0.20) + (gsc * 0.20) r = IAJResult(tap=tap, itd=itd, dpp=dpp, gsc=gsc, iaj=round(iaj, 2)) if itd < 30: r.octogono, r.razon = Octogono.ROJO_B, "ITD < 30%: sin devolución de resultados" elif iaj < 50 and tap < 15: r.octogono, r.razon = Octogono.ROJO_A, "Adultocentrismo activo: ignora demandas juveniles" elif iaj >= 85: r.octogono = Octogono.VERDE r.certificado_joven = True r.razon = "Certificado Joven activo — destacado en GOB.PE" r.gob_pe_url = f"https://www.gob.pe/institucion/{entidad_id}/certificado-joven" else: r.octogono, r.razon = Octogono.AMBAR, "En mejora — subir TAP e ITD para el verde" return r
